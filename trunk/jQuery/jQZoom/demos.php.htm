Example 1 - Standard Zoom

JAVASCRIPT (Mouseover the image.) 

var options =
            {
                zoomWidth: 250,
                zoomHeight: 250
            }
            $(".jqzoom").jqzoom(options);
Example 2 - Reverse Zoom

JAVASCRIPT (Mouseover the image.) 

            var options2 =
            {
                zoomWidth: 250,
                zoomHeight: 250,
                zoomType:'reverse'
            }

            $(".jqzoom2").jqzoom(options2);
Example 3 - Zoom without lens

JAVASCRIPT (Mouseover the image.) 

            var options3 =
            {
                zoomWidth: 200,
                zoomHeight: 200,
                xOffset: 20,
                title: false,
                lens:false

            }

            $(".jqzoom3").jqzoom(options3);
Example 4 - Custom positioning

JAVASCRIPT (Mouseover the image.) 

            var options4 =
            {

                zoomWidth: 300,
                zoomHeight: 200,
                position : 'right',
                yOffset :-100,
                xOffset :100,
                title :false


            }
            $(".jqzoom4").jqzoom(options4);
Example 5 - Fade out effect while hiding

JAVASCRIPT (Mouseover the image.) 

            var options5 =
            {
                zoomWidth: 250,
                zoomHeight: 250,
                showEffect:'show',
                hideEffect:'fadeout',
                fadeoutSpeed: 'slow',
                title :false
            }
            $(".jqzoom5").jqzoom(options5);
